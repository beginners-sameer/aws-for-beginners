#! /bin/bash
# Use this for your user data 
# install httpd ( Linux 2 Version )
yum update -y
yum install -y httpd
systemctl start httpd
systemctl enable httpd

echo "<h1> Hello AWS Engineers from $(hostname -f) in AZ $EC2_AVAIL_ZONE </h1>" > /var/www/html/index.html
